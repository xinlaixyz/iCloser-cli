## 问题

当前使用 readline 单行输入，Enter 提交，无法输入多行内容。Shift+Enter 无效果。

## 实现

### 文件: `src/cli/tui.ts` (扩展)

在已有 `setupRawInput` 基础上扩展：

```typescript
interface InputBox {
  lines: string[];           // 当前输入的行
  cursorRow: number;         // 光标所在行号
  cursorCol: number;         // 光标所在列号
  scrollOffset: number;      // 滚动偏移 (超出 6 行时)
  maxHeight: number;         // 最大可视高度 (6)
}

  // 标题 "输入"
  // 内容行 (最多显示 maxHeight 行)
  // 超出时顶部显示 ↑ 还有 N 行
  // 底部快捷键提示行
}

function handleInputKey(box: InputBox, ev: KeyEvent): InputAction {
  // Enter → 发送 (fire 'submit')
  // Shift+Enter → 插入 \n，拆分当前行
  // Backspace → 删除字符，处理行合并
  // Left/Right → 光标水平移动
  // Up/Down → 光标垂直移动 / 历史导航
  // Tab → 补全
  // 普通字符 → 插入
}
```

### 文件: `src/cli/repl.ts` (集成)

```typescript
// 替换 rl.on('line') 为 raw input handler
const inputBox = createInputBox();

setupRawInput(
  (ev) => {
    const action = handleInputKey(inputBox, ev);
    if (action.type === 'submit') {
      const text = inputBox.lines.join('\n');
      if (text.trim()) handleUserInput(text);
      resetInputBox(inputBox);
    }
    renderInputBox(inputBox);  // 重绘输入区域
  }
);
```
