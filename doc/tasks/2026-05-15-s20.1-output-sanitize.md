## 问题

AI stream 原始输出可能含 ANSI 转义码、控制字符、二进制残留，直接写入终端导致乱码。

## 实现

### 文件: `src/cli/output.ts`

新增 `sanitizeOutput(text: string): string` 函数：

```typescript
export function sanitizeOutput(text: string): string {
  let out = '';

  for (let i = 0; i < text.length; i++) {
    const code = text.charCodeAt(i);

    // Allow: \n(10), \r(13), \t(9), printable ASCII(32-126), Unicode above 127
    if (code === 10 || code === 13 || code === 9) { out += text[i]; continue; }
    if (code >= 32 && code <= 126) { out += text[i]; continue; }
    if (code > 127 && code < 65535) { out += text[i]; continue; }
    // Handle surrogate pairs (emoji etc.)
    if (code >= 0xD800 && code <= 0xDFFF) { out += text[i]; continue; }

    // Discard other control chars (0-8, 11-12, 14-31, 127)
  }

  // Collapse repeated non-ASCII junk
  out = out.replace(/(�){3,}/g, '�');
  out = out.replace(/[\x00-\x08\x0B\x0C\x0E-\x1F\x7F]/g, '');

  // Truncate overly long lines
  const lines = out.split('\n');
  out = lines.map(line => line.length > 1000 ? line.substring(0, 997) + '…(截断)' : line).join('\n');

  return out;
}
```
